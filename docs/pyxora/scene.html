260    def update(self,state: int):
261        """
262        @private
263        
264        (Not include in the doc as it should be call only inside scene itself)
265        Updates all the custom events and it's properties at the scene state.
266
267        Args:
268            state (int): The scene state. (default = -1, forever).

273            if not(self._is_state(state,event.state)):
274                continue
275
276            if not event.timer:
277                continue
278
279            if event.calls == event.loops:
280                continue

282            self._update_time(event)

284    def _now(self, state: int) -> float:
285        """Returns the current time based on the state.
286
287        Args:
288            state (int): Determines which time value to return.
289                - If 0: Returns runtime + pausetime.
290                - If >0: Returns runtime.
291                - If <0: Returns pausetime.
292
293        Returns:
294            float: The calculated current time value.
295        """
296        runtime = self.__scene.runtime
297        pausetime = self.__scene.pausetime
298        if state == 0:
299            return runtime + pausetime
300
301        now = runtime if state > 0 else pausetime
302        return now

305    def _update_time(self, event:"Event") -> None:
306        """Checks if an event should be triggered based on elapsed time.
307
308        If the time since the last event trigger exceeds the timer threshold,
309        the event is posted and its last_time is updated.
310
311        Args:
312            event (object): An object with the following attributes:
313                - state (int): State used to determine which time value to use.
314                - last_time (float): Timestamp of the last event trigger.
315                - timer (float): Timer threshold in milliseconds.
316                - name (str): Name of the event to post.
317
318        Side Effects:
319            Posts the event via self.post() if the condition is met.
320            Updates event.last_time.
321        """
322        now = self._now(event.state)
323        diff = (now - event.last_time) * 1000
324        is_time = diff >= event.timer
325        if not is_time:
326            return
327
328        self.post(event.name)
329        event.last_time = now
330        event.calls += 1

339            event_state (int): Current state of the event.

343        """
344        same = event_state == state

645    def __handle_events(self):
646        """Handles events during runtime or when paused."""
647        self.events.clear()
648        self.custom_events.clear()
649        self.Event.update(-1 if self.__paused else 1)
650
651        on_keydown, on_keyup, on_keypressed, on_wheel, on_event = self.__event_handlers[self.__paused]
652        for event in pygame.event.get():
653            self.events.add(event.type)
654
655            if hasattr(event, "custom"):
656                self.custom_events.add(event.name)
657                continue
658
659            if event.type == pygame.QUIT:
660                self.quit()
661
662            elif event.type == pygame.KEYDOWN:
663                key = pygame.key.name(event.key)
664                self.keys_pressed.add(key)
665                on_keydown(key)
666
667            elif event.type == pygame.KEYUP:
668                key = pygame.key.name(event.key)
669                self.keys_pressed.discard(key)
670                on_keyup(key)
671
672            elif event.type == pygame.MOUSEWHEEL:
673                wheel = "up" if event.y >= 1 else "down"
674                on_wheel(wheel)
675
676            elif event.type == pygame.VIDEORESIZE:
677                self.Display.resize((event.w, event.h))
678                # self.Camera.dynamic_zoom()
679            on_event(event)
680
681        if self.keys_pressed:
682            on_keypressed(self.keys_pressed)
