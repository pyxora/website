 79    def get(cls,source: str, *loc) -> Any:
 80        """
 81        Safely retrieve a nested value from a source dictionary.
 82
 83        Args:
 84            source (str): The data name to retrieve data from.
 85            *loc (str): A sequence of keys representing the path to the desired value.
 86
 87        Returns:
 88            Any: The value at the specified nested location, or None if the path is invalid.
 89
 90        Example:
 91            Assets.get("data"images", "player")  # Returns the player Surface if it exists
 92            Assets.get("engine,"images", "icon")  # Returns the engine icon Surface
 93        """
 94
 95        # return None if no location is provided
 96        # as having direct access to a dynamic attribute sounds scary lol
 97        if not loc:
 98            return None
 99
100        source = getattr(cls, source)
101        target = loc[0]
102
103        data = getattr(source, target, None)
104        if data is None:
105            return None
106
107        for key in loc[1:]:
108            if not isinstance(data, dict):
109                return None
110            data = data.get(key)
111
112        return data

115    def load(cls, source: "str") -> None:
116        """
117        Load file paths into the data system
118
119        Args:
120            source (str): The data name to retrieve data from.
121        """
122
123        data = getattr(cls, source)
124        for category, loader in loaders.items():
125            file_dict = data.files.get(category)
126            if not file_dict:
127                continue
128
129            asset_store = getattr(data, category)
130            for name, path in file_dict.items():
131                asset_store[name] = loader(path)

186    @staticmethod
187    def __get_all_files(path,ignore=None) -> dict[str, dict[str, str]]:
188        """
189        Recursively scan provided directories and build a nested dictionary of file paths.
190
191        Args:
192            path (dict[str, str]): A dictionary where each key is an asset type
193                                (like "images" or "fonts") and the value is the path to its folder.
194            ignore (set[str], optional): A set of folder names to exclude from scanning.
195                                        "__pycache__" is always ignored.
196
197        Returns:
198            dict[str, dict[str, str]]: A nested dictionary structured like:
199                {
200                    "images": {
201                        "player": "/path/to/images/player.png",
202                        "enemy": "/path/to/images/enemy.png"
203                    },
204                    "fonts": {
205                        "main": "/path/to/fonts/arial.ttf"
206                    },
207                    ...
208                }
209        """
210        if not ignore:
211            ignore = set()
212
213        ignore.update({"__pycache__"})
214
215        data = {}
216        for key,value in path.items():
217            for root, dirs, files in os.walk(value,topdown=False):
218                ftype = os.path.basename(root)
219                if ftype in ignore: continue
220                data[ftype] = {}
221                for file in files:
222                    full_path = os.path.join(root, file)
223                    name,_ = os.path.splitext(os.path.basename(full_path))
224                    data[ftype][name] = full_path
225        return data
226
227    @staticmethod
228    def __get_full_path(path):
229        """
230        Convert a relative path to an absolute normalized path and verify it exists.
231
232        Args:
233            path (str): The relative or partial path to validate.
234
235        Returns:
236            str: The normalized absolute path.
237
238        Raises:
239            OSError: If the resolved path does not exist.
240        """
241        path = os.path.normpath(os.getcwd()+path)
242        if not os.path.exists(path):
243            engine.error(OSError(f"The path doesn't exist: {path}"),debug)
244            engine.quit()
245        return path
